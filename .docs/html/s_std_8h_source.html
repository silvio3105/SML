864 template<typename T>
865 T str2Dec(const char* str, const char endChar = '\0')
866 {
867 T value = 0;
868 uint8_t idx = 0;
869
870 // Move to next char if first char is minus
871 if (str[0] == '-') idx++;
872
873 // While char is number and not equal to end char
874 while (str[idx] != endChar && sStd::isNum(str[idx]) == SSTD_OK)
875 {
876 // Add one digit to output value
877 value *= 10;
878
879 // Add converted char to output value
880 value += sStd::char2Num(str[idx]);
881
882 // Move index
883 idx++;
884 }
885
886 // If first char is minus, convert output value to negative
887 if (str[0] == '-') value *= (-1);
888
889 return value;
890 }
