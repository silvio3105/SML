105uint16_t sStd::sumDigits(T input)
106{
107 uint16_t sum = 0;
108
109 // Remove - sign if needed
110 SSTD_ABS(input);
111
112 // Do while input is not zero
113 do
114 {
115 // Increase sum with digit
116 sum += input % 10;
117
118 // Remove digit from input integer
119 input /= 10;
120 }
121 while (input);
122
123 // Return sum of all digits
124 return sum;

129char* sStd::tok(char* input, char separator)
130{
131 // Move pointer until separator is found or end of C-string is reached
132 while (*input && *input != separator) input++;
133
134 // If NULL character is found, return nullptr
135 if (!*input) return nullptr;
136
137 // Place NULL char over separator
138 *input = '\0';
139
140 // Move pointer to next character
141 input++;
142
143 // Return nullptr if character is NULL overwise return address of character
144 if (!*input) return nullptr;
145 else return input;
146}

160uint16_t sStd::count(const char* input, const char character, const char endChar)
161{
162 uint16_t count = 0;
163
164 // If selected character is not equal to end character
165 while (*input != endChar)
166 {
167 // If selected character is equal to wanted character, increas counter
168 if (*input == character) count++;

171 input++;

173
174 // Return number of character in input C-string

178uint8_t sStd::cmp(const char* input1, const char* input2, char endChar)
179{
180 // While both characters are not equal to endChar
181 while (*input1 != endChar)
182 {
183 // If characters are not equal return SSTD_NOK
184 if (*input1 != *input2) return SSTD_NOK;
185
186 // Move to next character
187 input1++;
188 input2++;
189 }
190
191 // Return SSTD_NOK if strings do not end with same character. This is input C-string length check
192 if (*input1 != *input2) return SSTD_NOK;
193
194 // Return equal status
195 return SSTD_OK;
196}

199// STRING SCAN FUNCTIONS DEFINITIONS
200uint8_t sStd::sscan(char* input, char sepBegin, uint8_t sepCntBegin, char sepEnd, uint8_t sepCntEnd, sStd::Data<char>& output, const uint8_t modify)
201{
202 // If first character is NULL
203 if (!*input)
204 {
205 // Set data to nullptr and 0 length and return not OK status
206 output.set(nullptr, 0);
207 return SSTD_NOK;
208 }
209
210 char* start = nullptr;
211
212 // Set start to first character if sepBegin param is NULL character
213 if (sepBegin) start = findToken(input, sepBegin, sepCntBegin, 1);
214 else start = input;
215
216 // If start of the token was not found
217 if (!start)
218 {
219 // Set data to nullptr and 0 length and return not OK status
220 output.set(nullptr, 0);
221 return SSTD_NOK;
222 }
223 else if (start[1]) // If next character is not \0
224 {
225 // Move start to next character
226 start++;
227
228 // Set output data to start address and end - start length
229 if (sepBegin) output.set(start, findToken(start, sepEnd, sepCntEnd, 0) - start);
230 else output.set(start - 1, findToken(start, sepEnd, sepCntEnd, 0) - start + 1); // -1 byte for start address abd +1 to include first character
231
232 // Replace character with \0 char
233 if (modify) output.get()[output.len()] = '\0';
234
235 return SSTD_OK;
236 }
237
238 return SSTD_NOK;
239}

241uint8_t sStd::sscan(char* input, sStd::scanData* data, const uint8_t len, const uint8_t modify, const uint8_t sorted)
242{
243 uint8_t total = 0;
244
245 // Loop through input list
246 for (uint8_t i = 0; i < len; i++)
247 {
248 // Call single sscan function.
249 if (sscan(input, sorted ? '\0' : data[i].sepBegin, data[i].sepCntBegin, data[i].sepEnd, data[i].sepCntEnd, data[i].output, modify) == SSTD_OK) total++;
250
251 // Calculate new start position if input list is sorted
252 if (sorted) input += data[i].output.len() + 1;
253 }
254
255 return total;
256}

389template<typename T, sStd::rbIdx_t N>
390uint8_t sStd::RingBuffer<T, N>::writeData(T data)
391{
392 // Return NOK status if no free data slots are available
393 if (!free()) return SSTD_NOK;
394
395 // Write data to head pointer
396 memory[head] = data;
397
398 // Move head pointer
399 head++;
400
401 // Increase new data counter
402 newCnt++;
403
404 // Reset head pointer
405 if (head == length) head = 0;
406
407 // Return OK status
408 return SSTD_OK;
409}

425// STATIC FUNCTION DEFINITIONS
426static char* sStd::findToken(char* input, char sep, char sepCnt, uint8_t retNull)
427{
428 // If character is not \0
429 while (*input)
430 {
431 // If wanted character is found
432 if (*input == sep)
433 {
434 // If that is wanted separator
435 if (!sepCnt) return input;
436 else sepCnt--;
437 }
438
439 // Move to next character
440 input++;
441 }
442
443 // Return nullptr if retNull param is not 0, otherwise return address of character
444 if (retNull) return nullptr;
445 else return input;
446}
