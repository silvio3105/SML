106// STRING MANIPULATION FUNCTIONS DEFINITIONS
107char* sStd::tok(char* input, char separator)
108{
109 // Move pointer until separator is found or end of C-string is reached
110 while (*input != '\0' && *input != separator) input++;
111
112 // If NULL character is found, return nullptr
113 if (!*input) return nullptr;
114
115 // Place NULL char over separator
116 *input = '\0';
117
118 // Move pointer to next character
119 input++;
120
121 // Return nullptr if character is NULL overwise return address of next character
122 if (!*input) return nullptr;
123 else return input;
124}

126uint16_t sStd::len(char* input, char endChar)
127{
128 // Copy string address to local variable
129 char* tmp = input;
130
131 // Move pointer until endChar is found
132 while (*tmp != endChar) tmp++;
133
134 // Return length of string.
135 return tmp - input;
136}

138uint16_t sStd::count(char* input, char character, char endChar)
139{
140 uint16_t count = 0;
141
142 // If selected character is not equal to end character
143 while (*input != endChar)
144 {
145 // If selected character is equal to wanted character, increas counter
146 if (*input == character) count++;
147
148 // Move to next character
149 input++;
150 }
151
152 // Return number of character in input C-string
153 return count;
154}

156uint8_t sStd::cmp(char* input1, char* input2, char endChar)
157{
158 // While both characters are not equal to endChar
159 while (*input1 != endChar)
160 {
161 // If characters are not equal return SSTD_NOK
162 if (*input1 != *input2) return SSTD_NOK;
163
164 // Move to next character
165 input1++;
166 input2++;
167 }
168
169 // Return SSTD_NOK if strings do not end with same character. This is input C-string length check
170 if (*input1 != *input2) return SSTD_NOK;
171
172 // Return equal status
173 return SSTD_OK;
174}

177// STRING SCAN FUNCTIONS DEFINITIONS
178uint8_t sStd::sscan(char* input, char sepBegin, uint8_t sepCntBegin, char sepEnd, uint8_t sepCntEnd, sStd::Data<char>& output)
179{
180 // If first character is NULL
181 if (!*input)
182 {
183 // Set data to nullptr and 0 length and return not OK status
184 output.set(nullptr, 0);
185 return SSTD_NOK;
186 }
187
188 char* start = nullptr;
189
190 // Set start to first character if sepBegin param is NULL character
191 if (sepBegin) start = findToken(input, sepBegin, sepCntBegin, 1);
192 else start = input;
193
194 // If start of the token was not found
195 if (!start)
196 {
197 // Set data to nullptr and 0 length and return not OK status
198 output.set(nullptr, 0);
199 return SSTD_NOK;
200 }
201 else if (start[1]) // If next character is not \0
202 {
203 // Move start to next character
204 start++;
205
206 // Set output data to start address and end - start length
207 if (sepBegin) output.set(start, findToken(start, sepEnd, sepCntEnd, 0) - start);
208 else output.set(start - 1, findToken(start, sepEnd, sepCntEnd, 0) - start + 1); // -1 byte for start address abd +1 to include first character
209 return SSTD_OK;
210 }
211
212 return SSTD_NOK;
213}

215uint8_t sStd::sscan(char* input, sStd::scanData* data, const uint8_t len)
216{
217 uint8_t total = 0;
218
219 // Loop through scan data list
220 for (uint8_t i = 0; i < len; i++)
221 {
222 // Call single sscan function.
223 if (sscan(input, data[i].sepBegin, data[i].sepCntBegin, data[i].sepEnd, data[i].sepCntEnd, data[i].output) == SSTD_OK) total++;
224 }
225
226 return total;
227

231// STATIC FUNCTION DEFINITIONS
232static char* sStd::findToken(char* input, char sep, char sepCnt, uint8_t retNull)
233{
234 // If character is not \0
235 while (*input)
236 {
237 // If wanted character is found
238 if (*input == sep)
239 {
240 // If that is wanted separator
241 if (!sepCnt) return input;
242 else sepCnt--;
243 }
244
245 // Move to next character
246 input++;
247 }
248
249 // Return nullptr if retNull param is not 0, otherwise return address of \c character
250 if (retNull) return nullptr;
251 else return input;
252}
