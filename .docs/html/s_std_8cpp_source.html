60char* sStd::tok(char* input, char separator)
61{
62 // Move pointer until separator is found or end of C-string is reached
63 while (*input && *input != separator) input++;
64
65 // If NULL character is found, return nullptr
66 if (!*input) return nullptr;
67
68 // Place NULL char over separator
69 *input = '\0';
70
71 // Move pointer to next character
72 input++;
73
74 // Return nullptr if character is NULL overwise return address of character
75 if (!*input) return nullptr;
76 else return input;
77}

109uint8_t sStd::cmp(const char* input1, const char* input2, char endChar)
110{
111 // While both characters are not equal to endChar
112 while (*input1 != endChar)
113 {
114 // If characters are not equal return SSTD_NOK
115 if (*input1 != *input2) return SSTD_NOK;
116
117 // Move to next character
118 input1++;
119 input2++;
120 }
121
122 // Return SSTD_NOK if strings do not end with same character. This is input C-string length check
123 if (*input1 != *input2) return SSTD_NOK;
124
125 // Return equal status
126 return SSTD_OK;
127}

153uint8_t sStd::validateDate(uint8_t day, uint8_t month, uint16_t year)
154{
155 // Validate year
156 if (year < 0 || year > 2100) return SSTD_NOK;
157
158 // Validate day
159 if (day < 0) return SSTD_NOK;
160
161 // Validate month and day
162 switch (month)
163 {
164 case 1:
165 case 3:
166 case 5:
167 case 7:
168 case 8:
169 case 10:
170 case 12:
171 {
172 if (day > 31) return SSTD_NOK;
173 break;
174 }
175
176 case 2:
177 {
178 // If is leap year
179 if (!(year % 4))
180 {
181 if (day > 29) return SSTD_NOK;
182 }
183 else // Not leap year
184 {
185 if (day > 28) return SSTD_NOK;
186 }
187 break;
188 }
189
190 case 4:
191 case 6:
192 case 9:
193 case 11:
194 {
195 if (day > 30) return SSTD_NOK;
196 break;
197 }
198
199 default: return SSTD_NOK;
200 }

202 return SSTD_OK;
203}

207uint8_t sStd::sscan(char* input, char sepBegin, uint8_t sepCntBegin, char sepEnd, uint8_t sepCntEnd, sStd::Data<char>& output, const uint8_t modify)
208{
209 // If first character is NULL
210 if (!*input)
211 {
212 // Set data to nullptr and 0 length and return not OK status
213 output.set(nullptr, 0);
214 return SSTD_NOK;
215 }
216
217 char* start = nullptr;
218
219 // Set start to first character if sepBegin param is NULL character
220 if (sepBegin) start = findToken(input, sepBegin, sepCntBegin, 1);
221 else start = input;
222
223 // If start of the token was not found
224 if (!start)
225 {
226 // Set data to nullptr and 0 length and return not OK status
227 output.set(nullptr, 0);
228 return SSTD_NOK;
229 }
230 else if (start[1]) // If next character is not \0
231 {
232 // Move start to next character
233 start++;
234
235 // Set output data to start address and end - start length
236 if (sepBegin) output.set(start, findToken(start, sepEnd, sepCntEnd, 0) - start);
237 else output.set(start - 1, findToken(start, sepEnd, sepCntEnd, 0) - start + 1); // -1 byte for start address abd +1 to include first character
238
239 // Replace character with \0 char
240 if (modify) output.get()[output.len()] = '\0';
241
242 return SSTD_OK;
243 }
244
245 return SSTD_NOK;
246}

248uint8_t sStd::sscan(char* input, sStd::scanData* data, const uint8_t len, const uint8_t modify, const uint8_t sorted)
249{
250 uint8_t total = 0;
251
252 // Loop through input list
253 for (uint8_t i = 0; i < len; i++)
254 {
255 // Call single sscan function.
256 if (sscan(input, sorted ? '\0' : data[i].sepBegin, data[i].sepCntBegin, data[i].sepEnd, data[i].sepCntEnd, data[i].output, modify) == SSTD_OK) total++;
257
258 // Calculate new start position if input list is sorted
259 if (sorted) input += data[i].output.len() + 1;
260 }
261
262 return total;
263}

267char* findToken(char* input, char sep, char sepCnt, const uint8_t retNull)
268{
269 // If character is not \0
270 while (*input)
271 {
272 // If wanted character is found
273 if (*input == sep)
274 {
275 // If that is wanted separator
276 if (!sepCnt) return input;
277 else sepCnt--;
278 }
279
280 // Move to next character
281 input++;
282 }
283
284 // Return nullptr if retNull param is not 0, otherwise return address of character
285 if (retNull) return nullptr;
286 else return input;
287}
