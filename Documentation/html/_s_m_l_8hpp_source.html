729 T str2Dec(const char* str, const char endChar = '\0')
730 {
731 T value = 0;
732 uint8_t idx = 0;
733
734 // Move to next char if first char is minus
735 if (str[0] == '-')
736 {
737 idx++;
738 }
739
740 // While char is number and not equal to end char
741 while (str[idx] != endChar && SML::isDigit(str[idx]) == SML::Answer_t::Yes)
742 {
743 // Add one digit to output value
744 value *= 10;
745
746 // Add converted char to output value
747 value += SML::char2Num(str[idx]);
748
749 // Move index
750 idx++;
751 }
752
753 // If first char is minus, convert output value to negative
754 if (str[0] == '-')
755 {
756 value *= (-1);
757 }
758
759 return value;

897 char* findToken(const char* input, const char sep, char sepCnt, const SML::Answer_t retNull)
898 {
899 // If character is not \0
900 while (*input)
901 {
902 // If wanted character is found
903 if (*input == sep)
904 {
905 // If that is wanted separator
906 if (!sepCnt)
907 {
908 return (char*)input;
909 }
910 else
911 {
912 sepCnt--;
913 }
914 }
915
916 // Move to next character
917 input++;
918 }
919
920 // Return nullptr if retNull param is not 0, otherwise return address of character
921 if (retNull == SML::Answer_t::Yes)
922 {
923 return nullptr;
924 }
925
926 return (char*)input;

965 SML::Return_t parse(char* input, char startSeparator, uint8_t startSeparatorCnt, char endSeparator, uint8_t endSeparatorCnt, SML::Data<char>& output, const SML::Answer_t modify = SML::Answer_t::No)
966 {
967 // If first character is NULL
968 if (!(*input))
969 {
970 // Set data to nullptr and 0 length and return not OK status
971 output.set(nullptr, 0);
972 return SML::Return_t::NOK;
973 }
974
975 char* start = nullptr;
976
977 // Set start to first character if startSeparator param is NULL character
978 if (startSeparator)
979 {
980 start = findToken(input, startSeparator, startSeparatorCnt, SML::Answer_t::Yes);
981 }
982 else
983 {
984 start = input;
985 }
986
987 // If start of the token was not found
988 if (!start)
989 {
990 // Set data to nullptr and 0 length and return not OK status
991 output.set(nullptr, 0);
992 return SML::Return_t::NOK;
993 }
994 else if (start[1]) // If next character is not \0
995 {
996 // Move start to next character
997 start++;
998
999 // Set output data to start address and end - start length
1000 if (startSeparator)
1001 {
1002 output.set(start, findToken(start, endSeparator, endSeparatorCnt, SML::Answer_t::No) - start);
1003 }
1004 else
1005 {
1006 output.set(start - 1, findToken(start, endSeparator, endSeparatorCnt, SML::Answer_t::No) - start + 1); // -1 byte for start address and +1 to include first character
1007 }
1008
1009 // Replace character with \0 char
1010 if (modify == SML::Answer_t::Yes)
1011 {
1012 output.get()[output.len()] = '\0';
1013 }
1014
1015 return SML::Return_t::OK;
1016 }
1017
1018 return SML::Return_t::NOK;
1019 }
