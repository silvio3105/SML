1071 {

1074
1075 // Move to next char if first char is minus

1091 idx++;

1096 {
1097 value *= (-1);
1098 }

1296 {

1298 if (!(*input))
1299 {
1300 // Set data to nullptr and 0 length and return not OK status
1301 output.set(nullptr, 0);
1302 return SML::Return_t::Nok;
1303 }
1304
1305 char* start = nullptr;
1306
1307 // Set start to first character if startSeparator param is NULL character
1308 if (startSeparator)
1309 {
1310 start = findToken(input, startSeparator, startSeparatorCnt, SML::Answer_t::Yes);
1311 }
1312 else
1313 {
1314 start = input;
1315 }
1316
1317 // If start of the token was not found
1318 if (!start)
1319 {
1320 // Set data to nullptr and 0 length and return not OK status
1321 output.set(nullptr, 0);
1322 return SML::Return_t::Nok;
1323 }
1324 else if (start[1]) // If next character is not \0
1325 {
1326 // Move start to next character
1327 start++;
1328
1329 // Set output data to start address and end - start length
1330 if (startSeparator)
1331 {
1332 output.set(start, findToken(start, endSeparator, endSeparatorCnt, SML::Answer_t::No) - start);
1333 }
1334 else
1335 {
1336 output.set(start - 1, findToken(start, endSeparator, endSeparatorCnt, SML::Answer_t::No) - start + 1); // -1 byte for start address and +1 to include first character
1337 }
1338
1339 // Replace character with \0 char
1340 if (modify == SML::Answer_t::Yes)
1341 {
1342 output.get()[output.len()] = '\0';
1343 }
1344
1345 return SML::Return_t::Ok;
1346 }
1347
1348 return SML::Return_t::Nok;
1349 }
