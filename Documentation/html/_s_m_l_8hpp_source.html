980 {
981 T value = 0;
982 uint8_t idx = 0;
983
984 // Move to next char if first char is minus
985 if (str[0] == '-')
986 {
987 idx++;
988 }

991 while (str[idx] != endChar && SML::isDigit(str[idx]) == SML::Answer_t::Yes)

994 value *= 10;
995

1005 {
1006 value *= (-1);
1007 }
1008

1204 SML::Return_t parse(char* input, char startSeparator, uint8_t startSeparatorCnt, char endSeparator, uint8_t endSeparatorCnt, SML::Data<char>& output, const SML::Answer_t modify)
1205 {
1206 // If first character is NULL
1207 if (!(*input))
1208 {
1209 // Set data to nullptr and 0 length and return not OK status
1210 output.set(nullptr, 0);
1211 return SML::Return_t::Nok;
1212 }
1213
1214 char* start = nullptr;
1215
1216 // Set start to first character if startSeparator param is NULL character
1217 if (startSeparator)
1218 {
1219 start = findToken(input, startSeparator, startSeparatorCnt, SML::Answer_t::Yes);
1220 }
1221 else
1222 {
1223 start = input;
1224 }
1225
1226 // If start of the token was not found
1227 if (!start)
1228 {
1229 // Set data to nullptr and 0 length and return not OK status
1230 output.set(nullptr, 0);
1231 return SML::Return_t::Nok;
1232 }
1233 else if (start[1]) // If next character is not \0
1234 {
1235 // Move start to next character
1236 start++;
1237
1238 // Set output data to start address and end - start length
1239 if (startSeparator)
1240 {
1241 output.set(start, findToken(start, endSeparator, endSeparatorCnt, SML::Answer_t::No) - start);
1242 }
1243 else
1244 {
1245 output.set(start - 1, findToken(start, endSeparator, endSeparatorCnt, SML::Answer_t::No) - start + 1); // -1 byte for start address and +1 to include first character
1246 }
1247
1248 // Replace character with \0 char
1249 if (modify == SML::Answer_t::Yes)
1250 {
1251 output.get()[output.len()] = '\0';
1252 }
1253
1254 return SML::Return_t::Ok;
1255 }
1256
1257 return SML::Return_t::Nok;
1258 }

1280 uint8_t parse(char* input, SML::parser_t* list, const uint8_t len, const SML::Answer_t modify, const SML::Answer_t sorted)
1281 {

1286 {
