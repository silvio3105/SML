669 return SML::Answer_t::No;

675 case 4:
676 case 6:

880 template<typename T>
881 T str2Dec(const char* str, const char endChar)
882 {
883 T value = 0;
884 uint8_t idx = 0;
885
886 // Move to next char if first char is minus
887 if (str[0] == '-')
888 {
889 idx++;
890 }
891
892 // While char is number and not equal to end char
893 while (str[idx] != endChar && SML::isDigit(str[idx]) == SML::Answer_t::Yes)
894 {
895 // Add one digit to output value
896 value *= 10;
897
898 // Add converted char to output value
899 value += SML::char2Num(str[idx]);
900
901 // Move index
902 idx++;
903 }
904
905 // If first char is minus, convert output value to negative
906 if (str[0] == '-')
907 {
908 value *= (-1);
909 }
910
911 return value;
912 }

1107 {
1108 // If first character is NULL
1109 if (!(*input))
1110 {
1111 // Set data to nullptr and 0 length and return not OK status
1112 output.set(nullptr, 0);
1113 return SML::Return_t::Nok;
1114 }
1115
1116 char* start = nullptr;
1117
1118 // Set start to first character if startSeparator param is NULL character
1119 if (startSeparator)
1120 {
1121 start = findToken(input, startSeparator, startSeparatorCnt, SML::Answer_t::Yes);
1122 }
1123 else
1124 {
1125 start = input;
1126 }
1127
1128 // If start of the token was not found
1129 if (!start)
1130 {
1131 // Set data to nullptr and 0 length and return not OK status
1132 output.set(nullptr, 0);
1133 return SML::Return_t::Nok;
1134 }
1135 else if (start[1]) // If next character is not \0
1136 {
1137 // Move start to next character
1138 start++;
1139
1140 // Set output data to start address and end - start length
1141 if (startSeparator)
1142 {
1143 output.set(start, findToken(start, endSeparator, endSeparatorCnt, SML::Answer_t::No) - start);
1144 }
1145 else
1146 {
1147 output.set(start - 1, findToken(start, endSeparator, endSeparatorCnt, SML::Answer_t::No) - start + 1); // -1 byte for start address and +1 to include first character
1148 }
1149
1150 // Replace character with \0 char
1151 if (modify == SML::Answer_t::Yes)
1152 {
1153 output.get()[output.len()] = '\0';
1154 }
1155
1156 return SML::Return_t::Ok;
1157 }
1158
1159 return SML::Return_t::Nok;
